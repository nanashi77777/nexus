### 1.1 ToolBatchMessage - 工具批次消息主体

```java
package cn.lin037.nexus.application.agent.model.dto;

import com.fasterxml.jackson.annotation.JsonFormat;
import lombok.Data;
import lombok.Builder;
import lombok.AllArgsConstructor;
import lombok.NoArgsConstructor;

import java.time.LocalDateTime;
import java.util.List;
import java.util.UUID;

/**
 * 工具批次消息 - 存储一次AI响应中的所有工具请求
 * 用于在AgentChatMessageEntity.acmContent中序列化存储
 */
@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
public class ToolBatchMessage {
    
    /**
     * 批次唯一标识符
     * 用于前端授权界面和后续操作的关联
     */
    private String batchId;
    
    /**
     * 关联的会话ID
     */
    private Long sessionId;
    
    /**
     * 关联的用户ID
     */
    private Long userId;
    
    /**
     * 关联的学习空间ID
     */
    private Long learningSpaceId;
    
    /**
     * 原始AI消息内容（不包含工具调用部分）
     * 用于显示AI在请求工具前的说明文字
     */
    private String aiMessageContent;
    
    /**
     * Token使用情况（序列化存储）
     */
    private TokenUsageInfo tokenUsage;
    
    /**
     * 工具请求列表
     * 保持与ToolExecutionRequest的顺序一致
     */
    private List<ToolRequestItem> toolRequests;
    
    /**
     * 批次整体状态
     */
    private ToolBatchStatus status;
    
    /**
     * 批次创建时间
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime createdAt;
    
    /**
     * 批次最后更新时间
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime updatedAt;
    
    /**
     * 批次完成时间（所有工具执行完毕或被取消）
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime completedAt;
    
    /**
     * 错误信息（批次级别的错误）
     */
    private String errorMessage;
    
    /**
     * 生成唯一的批次ID
     */
    public static String generateBatchId() {
        return "batch_" + UUID.randomUUID().toString().replace("-", "");
    }
    
    /**
     * 检查批次是否已完成（成功、失败或取消）
     */
    public boolean isCompleted() {
        return status == ToolBatchStatus.COMPLETED 
            || status == ToolBatchStatus.CANCELLED 
            || status == ToolBatchStatus.FAILED;
    }
    
    /**
     * 获取需要授权的工具数量
     */
    public long getPendingAuthCount() {
        return toolRequests.stream()
            .filter(item -> item.getStatus() == ToolRequestStatus.PENDING)
            .count();
    }
    
    /**
     * 获取已完成的工具数量
     */
    public long getCompletedCount() {
        return toolRequests.stream()
            .filter(item -> item.getStatus() == ToolRequestStatus.COMPLETED)
            .count();
    }
}
```

### 1.2 ToolRequestItem - 单个工具请求项

```java
package cn.lin037.nexus.application.agent.model.dto;

import com.fasterxml.jackson.annotation.JsonFormat;
import lombok.Data;
import lombok.Builder;
import lombok.AllArgsConstructor;
import lombok.NoArgsConstructor;

import java.time.LocalDateTime;

/**
 * 单个工具请求项
 * 包含工具的完整生命周期信息
 */
@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
public class ToolRequestItem {
    
    /**
     * 工具请求ID（对应ToolExecutionRequest.id()）
     * 用于与langchain4j的工具执行结果进行关联
     */
    private String requestId;
    
    /**
     * 工具名称（对应ToolExecutionRequest.name()）
     */
    private String toolName;
    
    /**
     * 工具参数（JSON字符串，对应ToolExecutionRequest.arguments()）
     */
    private String arguments;
    
    /**
     * 工具请求的当前状态
     */
    private ToolRequestStatus status;
    
    /**
     * 工具执行结果（JSON字符串）
     * 包含成功结果或错误信息
     */
    private String executionResult;
    
    /**
     * 执行错误信息（如果有）
     */
    private String errorMessage;
    
    /**
     * 工具执行耗时（毫秒）
     */
    private Long executionTimeMs;
    
    /**
     * 工具请求创建时间
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime requestedAt;
    
    /**
     * 工具开始执行时间
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime executionStartedAt;
    
    /**
     * 工具执行完成时间
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime executedAt;
    
    /**
     * 用户授权决策（APPROVE/REJECT/SKIP）
     * 仅在需要授权时使用
     */
    private String userDecision;
    
    /**
     * 用户授权时间
     */
    @JsonFormat(pattern = "yyyy-MM-dd HH:mm:ss")
    private LocalDateTime authorizedAt;
    
    /**
     * 授权用户ID（可能与会话用户不同，如管理员授权）
     */
    private Long authorizedByUserId;
    
    /**
     * 检查工具是否已执行完成
     */
    public boolean isExecuted() {
        return status == ToolRequestStatus.COMPLETED 
            || status == ToolRequestStatus.FAILED;
    }
    
    /**
     * 检查工具是否需要用户决策
     */
    public boolean needsUserDecision() {
        return status == ToolRequestStatus.PENDING;
    }
    
    /**
     * 从ToolExecutionRequest创建ToolRequestItem
     */
    public static ToolRequestItem fromToolExecutionRequest(dev.langchain4j.agent.tool.ToolExecutionRequest request) {
        return ToolRequestItem.builder()
            .requestId(request.id())
            .toolName(request.name())
            .arguments(request.arguments())
            .status(ToolRequestStatus.PENDING)
            .requestedAt(LocalDateTime.now())
            .build();
    }
}
```

```java
     * 从langchain4j的TokenUsage转换

            return null;

            .inputTokenCount(tokenUsage.inputTokenCount())
            .outputTokenCount(tokenUsage.outputTokenCount())
            .totalTokenCount(tokenUsage.totalTokenCount())
            .build();
    }
```

## 4. JSON序列化工具类

### 4.1 ToolMessageSerializer

```java
package cn.lin037.nexus.application.agent.util;

import cn.lin037.nexus.application.agent.model.dto.ToolBatchMessage;
import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.datatype.jsr310.JavaTimeModule;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;

/**
 * 工具消息序列化工具类
 * 负责ToolBatchMessage与JSON字符串之间的转换
 */
@Slf4j
@Component
public class ToolMessageSerializer {
    
    private final ObjectMapper objectMapper;
    
    public ToolMessageSerializer() {
        this.objectMapper = new ObjectMapper();
        this.objectMapper.registerModule(new JavaTimeModule());
        // 忽略未知属性，提高向后兼容性
        this.objectMapper.configure(
            com.fasterxml.jackson.databind.DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, 
            false
        );
    }
    
    /**
     * 将ToolBatchMessage序列化为JSON字符串
     * 
     * @param message 工具批次消息对象
     * @return JSON字符串
     * @throws RuntimeException 序列化失败时抛出
     */
    public String serialize(ToolBatchMessage message) {
        try {
            return objectMapper.writeValueAsString(message);
        } catch (JsonProcessingException e) {
            log.error("序列化ToolBatchMessage失败", e);
            throw new RuntimeException("工具批次消息序列化失败", e);
        }
    }
    
    /**
     * 将JSON字符串反序列化为ToolBatchMessage
     * 
     * @param json JSON字符串
     * @return 工具批次消息对象
     * @throws RuntimeException 反序列化失败时抛出
     */
    public ToolBatchMessage deserialize(String json) {
        try {
            return objectMapper.readValue(json, ToolBatchMessage.class);
        } catch (JsonProcessingException e) {
            log.error("反序列化ToolBatchMessage失败，JSON: {}", json, e);
            throw new RuntimeException("工具批次消息反序列化失败", e);
        }
    }
    
    /**
     * 安全的反序列化方法，失败时返回null
     * 
     * @param json JSON字符串
     * @return 工具批次消息对象，失败时返回null
     */
    public ToolBatchMessage deserializeSafely(String json) {
        try {
            return deserialize(json);
        } catch (Exception e) {
            log.warn("安全反序列化失败，返回null，JSON: {}", json, e);
            return null;
        }
    }
    
    /**
     * 验证JSON字符串是否为有效的ToolBatchMessage
     * 
     * @param json JSON字符串
     * @return 是否有效
     */
    public boolean isValidToolBatchMessage(String json) {
        return deserializeSafely(json) != null;
    }
}
```

## 6. 工具批次与消息实体的对应关系

### 6.1 存储策略

1. **一个ToolBatchMessage对应一个AgentChatMessageEntity**
    - `acmType = AgentChatMessageTypeEnum.TOOL_BATCH.getCode()`
    - `acmRole = AgentChatMessageRoleEnum.ASSISTANT.getRole()`
    - `acmContent = ToolBatchMessage的JSON序列化字符串`

2. **批次ID作为唯一标识**
    - 通过批次ID可以快速定位到对应的消息实体
    - 前端授权界面使用批次ID进行操作

3. **状态同步机制**
    - 工具执行状态变化时，更新ToolBatchMessage并重新序列化
    - 通过updateById更新AgentChatMessageEntity.acmContent

### 6.2 查询和更新流程

```java
// 示例：更新工具执行状态的完整流程

public void updateToolExecutionStatus(String batchId, String requestId, ToolRequestStatus newStatus, String result) {
    // 1. 查询消息实体
    AgentChatMessageEntity messageEntity = messageRepository.findToolBatchByBatchId(batchId)
        .orElseThrow(() -> new RuntimeException("工具批次不存在: " + batchId));
    
    // 2. 反序列化工具批次消息
    ToolBatchMessage batchMessage = toolMessageSerializer.deserialize(messageEntity.getAcmContent());
    
    // 3. 更新指定工具的状态
    ToolRequestItem targetTool = batchMessage.getToolRequests().stream()
        .filter(item -> requestId.equals(item.getRequestId()))
        .findFirst()
        .orElseThrow(() -> new RuntimeException("工具请求不存在: " + requestId));
    
    targetTool.setStatus(newStatus);
    targetTool.setExecutionResult(result);
    targetTool.setExecutedAt(LocalDateTime.now());
    
    // 4. 检查批次是否完成
    boolean allCompleted = batchMessage.getToolRequests().stream()
        .allMatch(item -> item.getStatus().isFinalState());
    
    if (allCompleted) {
        batchMessage.setStatus(ToolBatchStatus.COMPLETED);
        batchMessage.setCompletedAt(LocalDateTime.now());
    }
    
    batchMessage.setUpdatedAt(LocalDateTime.now());
    
    // 5. 重新序列化并保存
    messageEntity.setAcmContent(toolMessageSerializer.serialize(batchMessage));
    messageRepository.updateById(messageEntity);
}
```
